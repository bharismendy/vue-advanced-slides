mixins.js

					
export const mixin1 = {
	created() {
		console.log('Hello created - 1');
	}
}

export const mixin2 = {
	created() {
		console.log('Hello created - 2');
	}
}

export const mixin3 = {
	mounted() {
		console.log('Hello created - 3');
	}
}

export const mixin4 = {
	mounted() {
		console.log('Hello created - 4');
	}
}

OneComponent.vue

					
import {mixin1, mixin2, mixin3, mixin4} from './mixins.js';

export default {
	mixins: [mixin1, mixin2, mixin3, mixin4],

	created() {
		console.log('Created depuis le composant);
	},

	mounted() {
		console.log(`Mounted depuis le composant`);
	}
}

Resultat:

					
Hello created - 1
Hello created - 2
Created depuis le composant
Hello mounted - 3
Hello mounted - 4
